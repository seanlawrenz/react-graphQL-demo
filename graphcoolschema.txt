type File @model {
  contentType: String!
  createdAt: DateTime!
  id: ID! @isUnique
  name: String!
  secret: String! @isUnique
  size: Int!
  updatedAt: DateTime!
  url: String! @isUnique
}

type User @model {
  createdAt: DateTime!
  id: ID! @isUnique
  updatedAt: DateTime!
  webhookCreatedBy: WebhookConfig @relation(name: "CreatedByUser")
  webhookModifiedBy: WebhookConfig @relation(name: "ModifiedByUser")
}

type WebhookConfig @model {
  id: ID! @isUnique
  name: String!
  description: String
  payloadUrl: String!
  secret: String
  conentType: WebhookConfigContentType!
  sslVerificationEnabled: Boolean!
  allComponentEventsSelected: Boolean!
  isActive: Boolean!
  componentEventSelections: [WebhookConfigComponentEvent!]! @relation(name: "ComponentEventSelection")
  createdDate: DateTime!
  createdBy: User! @relation(name: "CreatedByUser")
  modifiedDate: DateTime!
  modifiedBy: User! @relation(name: "ModifiedByUser")
}

type WebhookConfigComponentEvent @model {
  id: ID! @isUnique
  webhookConfigComponentId: WebhookConfig @relation(name: "ComponentEventSelection")
  eventType: WebhookConfigEventType
  isSelected: Boolean!
}
